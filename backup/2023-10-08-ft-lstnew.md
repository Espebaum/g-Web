> ft_lstnew -- 새로운 연결 리스트를 할당하여 반환한다.


## (1) MY CODES
~~~c
#include "libft.h"

t_list	*ft_lstnew(void	*content)
{
	t_list	*new;

	new = (t_list *)malloc(sizeof(t_list));
	if (new == NULL)
		return (NULL);
	new->content = content;
	new->next = NULL;
	return (new);
}
~~~
